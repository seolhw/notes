28、封装求多个数组的交集

~~~javascript
function intersect(...args) {
  if (args.length === 0) {
    return [];
  }
  if (args.length === 1) {
    return args[0];
  }
  return args.reduce((prev, next) => {
    return prev.filter((i) => next.indexOf(i) > -1);
  });
}

~~~
